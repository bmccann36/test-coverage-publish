export async function createRepositories(): Promise<RepositoryContainer> {
  const ds = await getCachedDataSource();
  repositoryContainer = {
    syncBatchRepository: createErpBatchRespository(ds),
    invoicesRepository: createInvoicesRepositoryCustom(ds),
    paymentsRepository: createPaymentsRepositoryCustom(ds),
    erpSyncStatRepository: createErpSyncStatusRepository(ds),
    erpSettingsRepository: createErpSettingsRepositoryCustom(ds),
    integrationConfigRepository: createIntegrationConfigRepositoryCustom(ds),
    ibtRepository: createIbtRepositoryCustom(ds),
    yardRepository: createYardRepository(ds),
    finishedGoodRepository: createFinishedGoodRepository(ds),
