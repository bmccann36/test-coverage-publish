import {Column, Entity, Generated, Index, JoinColumn, ManyToOne, OneToMany} from 'typeorm';
import {ContactEntity} from './ContactEntity';
import {PriceSheetEntity} from './PriceSheetEntity';
import {YardEntity} from './YardEntity';
import {CustomerStatus} from './CustomerStatus';
import {InboundTicketEntity} from './InboundTicketEntity';
import {LocationEntity} from './LocationEntity';
import {OutboundTicketEntity} from './OutboundTicketEntity';
import {VehicleEntity} from './VehicleEntity';
 
@Index('customer_pkey', [ 'id' ], { unique: true })
@Entity('customer', { schema: 'public' })
export class CustomerEntity {
  @Column('uuid', { primary: true, name: 'id' })
  id: string;
 
  @Column('boolean', { name: 'is_supplier', default: () => 'false' })
  isSupplier: boolean;
 
  @Column('boolean', { name: 'is_consumer', default: () => 'false' })
  isConsumer: boolean;
 
  @Column('text', { name: 'display_name' })
  displayName: string;
 
  @Column('text', { name: 'company_name', nullable: true })
  companyName: string | null;
 
  @Column('enum', { name: 'type', enum: [ 'retail', 'industrial' ] })
  type: string;
 
  @Column('enum', {
    name: 'vendor_class',
    nullable: true,
    enum: [ 'demolition', 'construction', 'manufacturing', 'retail-preferred' ],
  })
  vendorClass: string;
 
  @Column('boolean', { name: 'archive', default: () => 'false' })
  archive: boolean;
 
  @Column('boolean', { name: 'buys_forbidden', nullable: true })
  buysForbidden: boolean | null;
 
  @Column('text', { name: 'notes', nullable: true })
  notes: string | null;
 
  @Column('text', { name: 'preferred_payment_method', nullable: true })
  preferredPaymentMethod?: string;
 
  @Column('text', { name: 'payment_terms', nullable: true })
  paymentTerms?: string;
 
  @Column('text', { name: 'default_currency', nullable: true })
  defaultCurrency?: string;
 
  @Column('numeric', {
    name: 'credit_balance',
    nullable: true,
    precision: 15,
    scale: 0,
  })
  creditBalance?: number;
 
  @Column('text', { name: 'credit_currency', nullable: true })
  creditCurrency?: string;
 
  @Generated('increment')
  @Column({ type: 'integer', name: 'external_id' })
  externalId: number | undefined;
 
  @Column('text', {name: 'erp_vendor_id'})
  erpVendorId: string;
 
  @Column('text', {name: 'erp_customer_id'})
  erpCustomerId: string;
 
  @ManyToOne(() => PriceSheetEntity, priceSheet => priceSheet.customers)
  @JoinColumn([{ name: 'price_sheet_id', referencedColumnName: 'id' }])
  priceSheet: PriceSheetEntity;
 
  @Column('uuid', { name: 'price_sheet_id' })
  priceSheetId: string;
 
  @ManyToOne(() => YardEntity, yard => yard.customers)
  @JoinColumn([{ name: 'yard_id', referencedColumnName: 'id' }])
  yard: YardEntity;
 
  @Column('uuid', { name: 'yard_id' })
  yardId: string;
 
  @OneToMany(() => CustomerStatus, customerStatus => customerStatus.customer)
  customerStatuses: CustomerStatus[];
 
  @OneToMany(() => InboundTicketEntity, inboundTicket => inboundTicket.customer)
  inboundTickets: InboundTicketEntity[];
 
  @OneToMany(() => LocationEntity, location => location.customer,
    {
      cascade: true,
    })
  locations: LocationEntity[];
 
  @OneToMany(() => OutboundTicketEntity, outboundTicket => outboundTicket.customer)
  outboundTickets: OutboundTicketEntity[];
 
  @OneToMany(() => VehicleEntity, vehicle => vehicle.customer,
    {
      cascade: true,
    })
  vehicles: VehicleEntity[];
 
  @OneToMany(() => ContactEntity, contact => contact.customer,
    {
      cascade: true,
    })
  contacts: ContactEntity[];
}
