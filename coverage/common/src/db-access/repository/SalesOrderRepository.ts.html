// noinspection TypeScriptValidateJSTypes
 
import {DataSource, In} from 'typeorm';
import {ErpToolSyncStatusEntity} from '../entity';
import {SalesOrderEntity} from '../entity';
import {QueryRunner, Repository} from 'typeorm';
import {UserOrgInfo} from '../../models';
 
 
export interface SalesOrderRepository {
  getIdsEligibleForSync(orgId: string, yardId: string): Promise<{ id: string }[]>;
 
  findValidSalesOrders(salesOrderIds: string[]): Promise<SalesOrderEntity[]>;
 
  deleteSalesOrder(salesOrderId: string): Promise<void>;
 
  getAllSalesOrderByYard(orgInfo: UserOrgInfo): Promise<SalesOrderEntity[]>;
 
  getByIdByYard(salesOrderId: string, orgInfo: UserOrgInfo): Promise<SalesOrderEntity | null>;
}
 
export const createSalesOrderRepositoryCustom = (ds: DataSource): SalesOrderRepository => {
  return ds.getRepository(SalesOrderEntity)
    .extend(<SalesOrderRepository>{
      //? getIdsEligibleForSync
      getIdsEligibleForSync(orgId: string, yardId: string): Promise<{ id: string; }[]> {
        const soQuery = ds.getRepository(SalesOrderEntity)
          .createQueryBuilder('salesOrder')
          .leftJoinAndSelect(ErpToolSyncStatusEntity, 'sync_stat', 'salesOrder.id = sync_stat.referencedEntityId')
          .where('salesOrder.yard_id = :yard_id', {yard_id: yardId})
          .andWhere('sync_stat.id ISNULL')
          .select([ 'salesOrder.id' ]);
        return soQuery.getMany();
      },
 
      findValidSalesOrders: async (salesOrderIds: string[]): Promise<SalesOrderEntity[]> => {
        return await ds.getRepository(SalesOrderEntity)
          .find({
            where: {id: In(salesOrderIds)},
            relations: {
              priceSheet: {
                priceEntries: {
                  materialType: true,
                },
              },
              customer: true,
              orderSalesDetails: true,
            },
          });
      },
 
      deleteSalesOrder: async (
        salesOrderId: string
      ): Promise<void> => {
        await ds.manager.transaction(async em => {
          await em.delete(SalesOrderEntity, {id: salesOrderId});
        });
      },
 
      getAllSalesOrderByYard: async (orgInfo: UserOrgInfo): Promise<SalesOrderEntity[]> => {
        const res = await ds.getRepository(SalesOrderEntity).find({
          where: {
            yardId: orgInfo.yardId,
          },
          relations: [ 'yard', 'priceSheet', 'orderSalesDetails', 'customer', 'customer.locations' ],
        });
        return res;
      },
 
      getByIdByYard: async (salesOrderId: string, orgInfo: UserOrgInfo): Promise<SalesOrderEntity | null> => {
        const res = await ds.getRepository(SalesOrderEntity).findOne({
          where: {
            id: salesOrderId,
            yardId: orgInfo.yardId,
          },
          relations: [ 'yard', 'priceSheet', 'orderSalesDetails', 'customer', 'customer.locations' ],
        });
        return res;
      },
    });
};
 
export const saveSalesOrder = async (
  salesOrderEntity: SalesOrderEntity, repo: Repository<SalesOrderEntity>
) => {
  await repo.save(salesOrderEntity);
};
 
export const getSalesOrder = async (id: string, repo: Repository<SalesOrderEntity>) => {
  return repo.findOne({
    where: {id},
  });
};
 
export const updateSalesOrderFulfilledAmount = (salesOrderTicketId: number, yardId: string, qr: QueryRunner) => {
  const query = `
      with result as (
          select p.id as price_sheet_id, sum(pe.fulfilled_weight) as fulfilled_amount
          from price_sheet p
                   join price_entry pe on p.id = pe.price_sheet_id
                   join sales_order so ON so.price_sheet_id = p.id
          where so.sales_order_ticket_id = $1
            and so.yard_id = $2
          group by p.id
      )
      update sales_order so
      set fulfilled_amount = result.fulfilled_amount
      from result
      WHERE so.price_sheet_id = result.price_sheet_id;`
 
  return qr.query(query, [ salesOrderTicketId, yardId ]);
}
 
export const updateSalesOrderPriceEntriesFulfilledAmounts = (salesOrderTicketId: number, yardId: string, qr: QueryRunner) => {
  const query = `update price_entry pe
                 set fulfilled_weight = result.net_weight
                 from (
                          with pe_to_update as (
                              select pe.material_type_id, so.sales_order_ticket_id, 0 as net_weight
                              from price_entry pe
                                       join price_sheet ps on pe.price_sheet_id = ps.id
                                       join sales_order so on ps.id = so.price_sheet_id
                              where sales_order_ticket_id = $2
                          ),
                               obt_material as (
                                   select m.material_type_id, ot.sales_order_ticket_id, sum(ms.net_weight) as net_weight
                                   from outbound_ticket ot
                                            JOIN material m on ot.id = m.outbound_ticket_id
                                            join material_status ms ON m.id = ms.material_id and ms.is_latest = true
                                   group by m.material_type_id, ot.sales_order_ticket_id
                               ),
                               obt_pl_material as (
                                   select m.material_type_id, ot.sales_order_ticket_id, sum(ms.net_weight)
                                   from outbound_ticket ot
                                            JOIN packing_list pl ON ot.id = pl.outbound_ticket_id
                                            JOIN finished_good fg ON pl.id = fg.packing_list_id
                                            JOIN material m on fg.id = m.finished_good_id
                                            JOIN material_status ms ON m.id = ms.material_id and ms.is_latest = true
                                   group by m.material_type_id, ot.sales_order_ticket_id
                               ),
                               obt_fg_material as (
                                   select m.material_type_id, ot.sales_order_ticket_id, sum(ms.net_weight)
                                   from outbound_ticket ot
                                            JOIN finished_good fg on ot.id = fg.outbound_ticket_id
                                            JOIN material m ON fg.id = m.finished_good_id
                                            JOIN material_status ms ON m.id = ms.material_id and ms.is_latest = true
                                   group by m.material_type_id, ot.sales_order_ticket_id
                               ),
                               obt_materials as (select material_type_id,
                                                        sales_order_ticket_id,
                                                        sum(net_weight) as net_weight
                                                 from (
                                                          select *
                                                          from pe_to_update
                                                          union
                                                          select *
                                                          from obt_material
                                                          union
                                                          select *
                                                          from obt_pl_material
                                                          union
                                                          select *
                                                          from obt_fg_material) tmp
                                                 group by material_type_id, sales_order_ticket_id)
-- join sales order to outbound ticket through price entry
                          select pe.id as price_entry_id, om.net_weight
                          from price_entry pe
                                   JOIN price_sheet ps ON ps.id = pe.price_sheet_id
                                   JOIN sales_order so on ps.id = so.price_sheet_id
                                   JOIN obt_materials om ON om.sales_order_ticket_id = so.sales_order_ticket_id and
                                                            pe.material_type_id = om.material_type_id
 
                          where pe.yard_id = $1
                            and so.sales_order_ticket_id = $2
                      ) result
                 where result.price_entry_id = pe.id;`
 
  return qr.query(query, [ yardId, salesOrderTicketId ]);
}
