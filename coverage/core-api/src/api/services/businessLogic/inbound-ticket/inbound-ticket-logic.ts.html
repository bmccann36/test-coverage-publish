export async function removeFailedImages(inboundTicket: InboundTicketRequestDTO | InboundTicketUpdateRequestDTO, orgInfo: UserOrgInfo) {
  if (inboundTicket?.status?.image && inboundTicket?.status?.imageIsTemp) {
    await awaitOrgImage(inboundTicket?.status?.image, orgInfo)
      .catch(err => {
        console.error('inbound ticket gross image', inboundTicket?.status?.image, 'failed to upload to S3', err);
        inboundTicket.status.image = undefined;
        inboundTicket.status.imageIsTemp = undefined;
      });
  }
  if (inboundTicket?.status?.tareImage && inboundTicket?.status?.tareImageIsTemp) {
    await awaitOrgImage(inboundTicket?.status?.tareImage, orgInfo)
      .catch(err => {
        console.error('inbound ticket tare image', inboundTicket?.status?.tareImage, 'failed to upload to S3', err);
        inboundTicket.status.tareImage = undefined;
        inboundTicket.status.tareImageIsTemp = undefined;
      });
  }
  const materialsWithTempImages = inboundTicket?.materials
    ?.filter(m => (m.image && m.imageIsTemp) || (m.tareImage && m.tareImageIsTemp));
 
  if (!materialsWithTempImages) return;
 
  await mapAsync(materialsWithTempImages, async m => {
    if (m.image) {
      await awaitOrgImage(m.image, orgInfo)
        .catch(err => {
          console.error('material gross image', m.image, 'failed to upload to S3', err);
          m.image = undefined;
          m.imageIsTemp = undefined;
        });
    }
    if (m.tareImage) {
      await awaitOrgImage(m.tareImage, orgInfo)
        .catch(err => {
          console.error('material tare image', m.tareImage, 'failed to upload to S3', err);
          m.tareImage = undefined;
          m.tareImageIsTemp = undefined;
        });
    }
  });
}
