export async function mapPaymentToQbBillPayment(ibtList: InboundTicketEntity[], payment: PartialPayment, qboSettings: QboYardSettings): Promise<QboBillPayment> {
  const syncStatusRepo = repositoryContainer.erpSyncStatRepository;
 
  const ibtIds = ibtList.map((ibt: InboundTicketEntity) => {
    return ibt.id;
  });
  /* get all ErpToolSyncStatusEntity for ibts
   and match with ibt.id from partial ibts we need to know the referenced
    billID in quickbooks in order to create a payment that is related to the
     previously created bill in quickbooks */
  const billsForTxn: ErpToolSyncStatusEntity[] =
    await syncStatusRepo.getSyncStatusByEntityTypeAndId(ibtIds, ENTITY_TYPE.INBOUND_TICKET);
 
  const paymentMethod = PAY_TYPE.CHECK;
  const checkPayment: CheckPayment = {
    PrintStatus: 'NeedToPrint',
    BankAccountRef: {
      name: qboSettings.checkAccountRef.name,
      value: qboSettings.checkAccountRef.value,
    },
  };
 
  // if customer is registered in quickbooks already, take the erpId.
  // else take the default customer
  let vendorRef;
  let paymentNote;
  if (ibtList[0].customer.erpVendorId) {
    vendorRef = ibtList[0].customer.erpVendorId; // declared at IBT level
    paymentNote = undefined;
  }
  else {
    console.log(
      `Payment with id, ${payment.id}, does not have erpVendorId on customer. Using Default Greenspark Customer`
    );
    vendorRef = qboSettings.defaultVendor.value;
    paymentNote = ibtList[0].customer.displayName;
  }
 
  return {
    VendorRef: {
      value: vendorRef, // declared at IBT level
    },
    Line: mapIbtListToLineItemForBillPayment(ibtList, billsForTxn),
    TotalAmt: payment.total,
    PayType: paymentMethod,
    CurrencyRef: {
      name: 'United States Dollar',
      value: 'USD',
    },
    PrivateNote: paymentNote, // undefined if vendor exists in qbo, else GS customer
    TxnDate: new Date(payment.paymentDate).toISOString(),
    MetaData: {
      CreateTime: new Date(payment.paymentDate).toISOString(),
      LastUpdatedTime: new Date(payment.completionDate).toISOString(),
    },
    CheckPayment: checkPayment,
  };
 
}
 
export async function mapPaymentToQbPurchase(ibtList: InboundTicketEntity[], payment: PartialPayment, yardQboSettings: QboYardSettings): Promise<QboPurchase> {
  const syncStatusRepo = repositoryContainer.erpSyncStatRepository;
 
  const ibtIds = ibtList.map((ibt: InboundTicketEntity) => {
    return ibt.id;
  });
  /* get all ErpToolSyncStatusEntity for ibts and match with ibt.id from partial ibts we need to know the referenced billID in quickbooks in order to create a payment that is related to the previously created bill in quickbooks */
  const billsForTxn: ErpToolSyncStatusEntity[] =
    await syncStatusRepo.getSyncStatusByEntityTypeAndId(ibtIds, ENTITY_TYPE.INBOUND_TICKET);
 
  // customer on ibt will always match customer on payment so it's ok to just grab ibt 0
  const vendorName = ibtList[0].customer?.displayName;
 
  const paymentMethod = PAY_TYPE.CASH;
  const account = {
    name: yardQboSettings.checkAccountRef.name,
    value: yardQboSettings.checkAccountRef.value,
  };
 
  return {
    AccountRef: account, // needs to be a bank account
    Line: mapIbtListToLineItemForPurchase(ibtList, billsForTxn),
    PaymentType: paymentMethod,
    CurrencyRef: {
      name: 'United States Dollar',
      value: 'USD',
    },
    // rather than show notes related to the purchase we just display the vendor associated with the purchase
    PrivateNote: vendorName,
    TxnDate: new Date(payment.paymentDate).toISOString(),
    MetaData: {
      CreateTime: new Date(payment.paymentDate).toISOString(),
      LastUpdatedTime: new Date(payment.completionDate).toISOString(),
    },
    DocNumber: `${GRNSPK_PREFIX}${payment.paymentNumber}`,
  };
}
 
function mapIbtListToLineItemForPurchase(ibtList: InboundTicketEntity[], billsForTxn: ErpToolSyncStatusEntity[]): PurchaseLineItem[] {
 
  const lineItems: any[] = ibtList.map((ibt: InboundTicketEntity) => {
 
    for (const item of billsForTxn) {
      if (item.referencedEntityId == ibt.id) {
 
        const materialItemArray = ibt.materials?.map(material => {
          return {
            DetailType: 'ItemBasedExpenseLineDetail',
            Amount: material.materialStatusVw?.netCost, // comes from latest material status
            ItemBasedExpenseLineDetail: {
              Qty: String(material.materialStatusVw?.netWeight),
              ItemRef: { // comes from material_type we store the mapping to QB on our side
                value: material.materialType?.erpItemId,
              },
            },
            LinkedTxn: [{
              TxnId: item.erpItemId,
              TxnType: 'Bill',
            }],
          };
        });
 
        return materialItemArray;
      }
    }
  });
 
  return lineItems.flat(1);
}
